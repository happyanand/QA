package Selenium;

import java.util.List;

import org.openqa.selenium.By;
import org.openqa.selenium.WebDriver;
import org.openqa.selenium.WebElement;
import org.openqa.selenium.chrome.ChromeDriver;
import org.openqa.selenium.support.ui.Select;

public class Dropdown 
{
	public static void main(String[] args) throws InterruptedException 
	{
		System.setProperty("webdriver.chrome.driver", "./Software/chromedriver.exe");
		WebDriver driver=new ChromeDriver();
		Thread.sleep(1000);
		driver.manage().window().maximize();
		driver.get("file:///D:/QA/MultiSelect.html");
		Thread.sleep(2000);
		WebElement ele = driver.findElement(By.id("Menu"));
		ele.click();
		Select s=new Select(ele);
		boolean b = s.isMultiple();
		System.out.println(b);
        s.selectByValue("Gobi");
        s.selectByIndex(2);
        s.selectByVisibleText("Noodalls");
        WebElement fs = s.getFirstSelectedOption();
        System.out.println(fs.getText());
        List<WebElement> as = s.getAllSelectedOptions();
        int size=as.size();
        for(int i=0;i<size;i++)
        {
        	String se=as.get(i).getText();
        	System.out.println(se);
        }
		Thread.sleep(2000);
		s.deselectAll();
		Thread.sleep(2000);
		List<WebElement> op=s.getOptions();
		int size1=op.size();
		for(int i=0;i<size1;i++)
		{
			String opt=op.get(i).getText();
			System.out.println(opt);
		}
		System.out.println(s.getWrappedElement().getText());
		driver.quit();
	}
}

package Selenium;

import org.openqa.selenium.By;

import org.openqa.selenium.WebElement;
import org.openqa.selenium.chrome.ChromeDriver;

public class Sendkeys_Keyboards 
{
	public static void main(String[] args) throws InterruptedException 
	{
		
		System.setProperty("webdriver.chrome.driver", "./Software/chromedriver.exe");
		WebDriver driver=new ChromeDriver();
		Thread.sleep(1000);
		driver.manage().window().maximize();
		driver.get("file:///D:/QA/Selenium/Keys.html");
		Thread.sleep(2000);
		WebElement ele = driver.findElement(By.xpath("//input[@values='Anand']"));
		ele.sendKeys("Anandhakumar");
		Thread.sleep(2000);
		ele.sendKeys(Keys.ENTER);
		Thread.sleep(2000);
		ele.sendKeys(Keys.CONTROL+"a");
		Thread.sleep(2000);
		ele.sendKeys(Keys.CONTROL+"c");
		Thread.sleep(2000);
		WebElement ele1 = driver.findElement(By.xpath("//input[@values='kumar']"));
		ele1.sendKeys(Keys.CONTROL+"av");

}

package Selenium;

import java.awt.AWTException;

import org.openqa.selenium.WebDriver;
import org.openqa.selenium.edge.EdgeDriver;

public class Navigation 
{
	public static void main(String[] args) throws InterruptedException, AWTException 
	{
		System.setProperty("webdriver.edge.driver", "./Software/msedgedriver.exe");
		WebDriver driver=new EdgeDriver();
		Thread.sleep(1000);
		driver.manage().window().maximize();
		driver.get("https://www.google.com");
		Thread.sleep(2000);
		System.out.println(driver.getTitle());
		driver.navigate().to("https://reqres.in/");
		Thread.sleep(2000);
		System.out.println(driver.getTitle());
		driver.navigate().back();
		Thread.sleep(2000);
		System.out.println(driver.getTitle());
		driver.navigate().refresh();
		Thread.sleep(2000);
		driver.navigate().forward();
		System.out.println(driver.getTitle());
		Thread.sleep(2000);
		driver.close();
	}
}


//Robot Keys: Lauch from Microsoft Edge

package Selenium;

import java.awt.AWTException;
import java.awt.Robot;
import java.awt.event.KeyEvent;

import org.openqa.selenium.WebDriver;
import org.openqa.selenium.edge.EdgeDriver;

public class Robot_Key_MicrosoftEdge 
{
	public static void main(String[] args) throws InterruptedException, AWTException 
	{
		System.setProperty("webdriver.edge.driver", "./Software/msedgedriver.exe");
		WebDriver driver=new EdgeDriver();
		Thread.sleep(1000);
		driver.manage().window().maximize();
		driver.get("https://www.google.com");
		Thread.sleep(2000);
		Robot robot = new Robot();
		robot.keyPress(KeyEvent.VK_CONTROL);
		robot.keyPress(KeyEvent.VK_T);
		robot.keyRelease(KeyEvent.VK_CONTROL);
		robot.keyRelease(KeyEvent.VK_T);
		Thread.sleep(2000);
		driver.navigate().to("https://reqres.in/");
		Thread.sleep(2000);
		driver.close();
	}
}


//JavaScriptExecution:(ScrollBy):

package Selenium;

import java.io.IOException;

import org.apache.poi.EncryptedDocumentException;
import org.openqa.selenium.JavascriptExecutor;
import org.openqa.selenium.WebDriver;
import org.openqa.selenium.chrome.ChromeDriver;

public class Scrollby 
{
	public static void main(String[] args) throws InterruptedException, EncryptedDocumentException, IOException 
	{
		System.setProperty("webdriver.chrome.driver", "./Software/chromedriver.exe");
		WebDriver driver=new ChromeDriver();
		Thread.sleep(1000);
		driver.manage().window().maximize();
		driver.get("https://reqres.in/");
		Thread.sleep(2000);
		JavascriptExecutor js = (JavascriptExecutor) driver;
		for(int i=0;i<15;i++)
		{
			js.executeScript("window.scrollBy(0,500)");
			Thread.sleep(500);
		}
		for(int j=0;j<15;j++)
		{
			js.executeScript("window.scrollBy(0,-500)");
			Thread.sleep(500);
		}
		driver.close();
	}
}

import org.openqa.selenium.WebDriver;

public class Assert1 
{
	@Test

	{
